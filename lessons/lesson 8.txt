    //////  LESSON #8: CREATING ASTEROIDS ON A TIMER  //////
    //////                DIAGONAL                    //////

    var i = 0;
    var moveInterval = setInterval(function(){
		if(i < 5) {
			new Orb(1.0)
				.translateTo(100*i,100*i,0)
			i++;
		}
		else {
			clearInterval(moveInterval);
		}
    }, 500);

    //The code above creates asteroids on a timer in a diagonal pattern
